PowerPC Assembly
Hello World Example:

# Hello World for Xbox 360 PPC ASM.
# Addresses are kernel-dependent, and the code to get them is
# long, so I have chosen to omit it and insert placeholders.

.global _start
.text

_start:
  # Align the stack onto a 16-byte boundary
  lis r3, 0xFFFF
  ori r3, r3, 0xFFF0
  and r1, r1, r3

  # Load the address of DbgPrint into CTR 
  lis r3, <high half of DbgPrint address>
  ori r3, r3, <low half of DbgPrint address>
  mtctr r3
  
  # Load the address of the hello world string into r3 (arg 1)
  lis r3, hello_world@ha
  ori r3, r3, hello_world@l
  
  # Clear the other arg registers
  li r4, 0
  li r5, 0
  li r6, 0
  li r7, 0
  li r8, 0
  li r9, 0
  li r10, 0
  
  # Create a new stack frame for the function call
  subi r1, r1, 0x60
  
  # Execute DbgPrint (branch to CTR)
  bctrl
  
  # Pop the stack frame we created
  addi r1, r1, 0x60

  # Load the address of XamLoaderLaunchTitle into CTR
  lis r3, <high half of XamLoaderLaunchTitle address>
  ori r3, r3, <low half of XamLoaderLaunchTitle address>
  mtctr r3
  
  # Clear all arg registers
  li r3, 0
  li r4, 0
  li r5, 0
  li r6, 0
  li r7, 0
  li r8, 0
  li r9, 0
  li r10, 0
  
  # Create a new stack frame
  subi r1, r1, 0x60
  
  # Execute XamLoaderLaunchTitle (return to dashboard)
  bctr
  
.data

hello_world:
  .asciz "Hello, World!\n"
